  const createModal = () => {
    if (document.getElementById('creditsModal')) {
      return;
    }

    const modal = document.createElement('div');
    modal.id = 'creditsModal';
    modal.className = 'credits-modal';

    let creditsListHTML = '';
    creditsList.forEach(credit => {
      if (typeof credit === 'string') {
        creditsListHTML += `<li>${credit}</li>`;
      } else {
        creditsListHTML += `
                    <li>
                        <div class="credit-item-with-image">
                            <img src="${credit.image}" alt="${credit.name}" class="credit-avatar">
                            <div class="credit-text">
                                <span class="credit-name">${credit.name}</span>, ${credit.text}
                            </div>
                        </div>
                    </li>
                `;
      }
    });

    modal.innerHTML = `
            <div class="credits-content">
                <span class="close-credits">&times;</span>
                <h3>CREDITS</h3>
                <ul>${creditsListHTML}</ul>
            </div>
        `;

    document.body.appendChild(modal);

    document.querySelector('.close-credits').addEventListener('click', hideCredits);
    modal.addEventListener('click', event => {
      if (event.target === modal) {
        hideCredits();
      }
    });
  };

    createModal();
    document.getElementById('creditsModal').style.display = 'flex';
  };

  const hideCredits = () => {
    const modal = document.getElementById('creditsModal');
    if (modal) {
      modal.style.display = 'none';
    }
  };

  const setupKeyListener = () => {
    document.addEventListener('keydown', event => {
      if (event.key === 'Escape') {
        hideCredits();
      }
    });
